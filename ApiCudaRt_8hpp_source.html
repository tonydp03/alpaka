81  // Host function helper
82  // Encapsulates the different function signatures used by cudaStreamAddCallback and cudaLaunchHostFn, and the
83  // different calling conventions used by CUDA (__stdcall on Win32) and HIP (standard).
84  struct HostFnAdaptor
85  {
86  HostFn_t func_;
87  void* data_;
88 
89  static void CUDART_CB hostFunction(void* data)
90  {
91  auto ptr = reinterpret_cast<HostFnAdaptor*>(data);
92  ptr->func_(ptr->data_);
93  delete ptr;
94  }
95 
96  static void CUDART_CB streamCallback(Stream_t, Error_t, void* data)
97  {
98  auto ptr = reinterpret_cast<HostFnAdaptor*>(data);
99  ptr->func_(ptr->data_);
100  delete ptr;
101  }
102  };

237  static inline Error_t launchHostFunc(Stream_t stream, HostFn_t fn, void* userData)
238  {
239 # if BOOST_LANG_CUDA >= BOOST_VERSION_NUMBER(10, 0, 0)
240  // Wrap the host function using the proper calling convention
241  return ::cudaLaunchHostFunc(stream, HostFnAdaptor::hostFunction, new HostFnAdaptor{fn, userData});
242 # else
243  // Emulate cudaLaunchHostFunc using cudaStreamAddCallback with a callback adaptor.
244  return ::cudaStreamAddCallback(stream, HostFnAdaptor::streamCallback, new HostFnAdaptor{fn, userData}, 0);
245 # endif
246  }
