60  //! The CUDA/HIP Xor random number generator engine.
61  class Xor
62  {
63  public:
64  // After calling this constructor the instance is not valid initialized and
65  // need to be overwritten with a valid object
66 # if defined(ALPAKA_ACC_GPU_CUDA_ENABLED)
67  ALPAKA_FN_HOST_ACC Xor() : state(curandStateXORWOW_t{})
68 # else
69  ALPAKA_FN_HOST_ACC Xor() : state(hiprandStateXORWOW_t{})
70 # endif
71  {
72  }
73 
74  __device__ Xor(
75  std::uint32_t const& seed,
76  std::uint32_t const& subsequence = 0,
77  std::uint32_t const& offset = 0)
78  {
79 # ifdef ALPAKA_ACC_GPU_CUDA_ENABLED
80  curand_init(seed, subsequence, offset, &state);
81 # else
82  hiprand_init(seed, subsequence, offset, &state);
83 # endif
84  }
85 
86  public:
87 # ifdef ALPAKA_ACC_GPU_CUDA_ENABLED
88  curandStateXORWOW_t state;
89 # else
90  hiprandStateXORWOW_t state;
91 # endif
92 
93  // STL UniformRandomBitGenerator concept. This is not strictly necessary as the distributions
94  // contained in this file are aware of the API specifics of the CUDA/HIP XORWOW engine and STL
95  // distributions might not work on the device, but it servers a compatibility bridge to other
96  // potentially compatible alpaka distributions.
97 # ifdef ALPAKA_ACC_GPU_CUDA_ENABLED
98  using result_type = decltype(curand(&state));
99 # else
100  using result_type = decltype(hiprand(&state));
101 # endif
102  ALPAKA_FN_HOST_ACC constexpr static result_type min()
103  {
104  return std::numeric_limits<result_type>::min();
105  }
106  ALPAKA_FN_HOST_ACC constexpr static result_type max()
107  {
108  return std::numeric_limits<result_type>::max();
109  }
110  __device__ result_type operator()()
111  {
112 # ifdef ALPAKA_ACC_GPU_CUDA_ENABLED
113  return curand(&state);
114 # else
115  return hiprand(&state);
116 # endif
117  }
118  };

164  //! The CUDA/HIP random number float uniform distribution.
165  template<>
166  class UniformReal<float>
167  {
168  public:
169  template<typename TEngine>
170  __device__ auto operator()(TEngine& engine) -> float
171  {
172  // (0.f, 1.0f]
173 # ifdef ALPAKA_ACC_GPU_CUDA_ENABLED
174  float const fUniformRand(curand_uniform(&engine.state));
175 # else
176  float const fUniformRand(hiprand_uniform(&engine.state));
177 # endif
178  // NOTE: (1.0f - curand_uniform) does not work, because curand_uniform seems to return
179  // denormalized floats around 0.f. [0.f, 1.0f)
180  return fUniformRand * static_cast<float>(fUniformRand != 1.0f);
181  }
182  };
