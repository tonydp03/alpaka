81  //! The CUDA/HIP RT sync queue enqueue trait specialization.
82  template<typename TTask>
83  struct Enqueue<QueueUniformCudaHipRtNonBlocking, TTask>
84  {
85  enum class CallbackState
86  {
87  enqueued,
88  notified,
89  finished,
90  };
91 
92  struct CallbackSynchronizationData : public std::enable_shared_from_this<CallbackSynchronizationData>
93  {
94  std::mutex m_mutex;
95  std::condition_variable m_event;
96  CallbackState state = CallbackState::enqueued;
97  };
98 
99 # if defined(ALPAKA_ACC_GPU_CUDA_ENABLED)
100  static void CUDART_CB
101 # else
102  static void HIPRT_CB
103 # endif
104  uniformCudaHipRtCallback(
105  ALPAKA_API_PREFIX(Stream_t) /*queue*/,
106  ALPAKA_API_PREFIX(Error_t) /*status*/,
107  void* arg)
108  {
109  // explicitly copy the shared_ptr so that this method holds the state even when the executing thread
110  // has already finished.
111  const auto pCallbackSynchronizationData
112  = reinterpret_cast<CallbackSynchronizationData*>(arg)->shared_from_this();
113 
114  // Notify the executing thread.
115  {
116  std::unique_lock<std::mutex> lock(pCallbackSynchronizationData->m_mutex);
117  pCallbackSynchronizationData->state = CallbackState::notified;
118  }
119  pCallbackSynchronizationData->m_event.notify_one();
120 
121  // Wait for the executing thread to finish the task if it has not already finished.
122  std::unique_lock<std::mutex> lock(pCallbackSynchronizationData->m_mutex);
123  if(pCallbackSynchronizationData->state != CallbackState::finished)
124  {
125  pCallbackSynchronizationData->m_event.wait(
126  lock,
127  [pCallbackSynchronizationData]()
128  { return pCallbackSynchronizationData->state == CallbackState::finished; });
129  }
130  }
131 
132  ALPAKA_FN_HOST static auto enqueue(QueueUniformCudaHipRtNonBlocking& queue, TTask const& task) -> void
133  {
134  auto pCallbackSynchronizationData = std::make_shared<CallbackSynchronizationData>();
135  ALPAKA_UNIFORM_CUDA_HIP_RT_CHECK(ALPAKA_API_PREFIX(StreamAddCallback)(
136  queue.getNativeHandle(),
137  uniformCudaHipRtCallback,
138  pCallbackSynchronizationData.get(),
139  0u));
140 
141  // We start a new std::thread which stores the task to be executed.
142  // This circumvents the limitation that it is not possible to call CUDA methods within the CUDA/HIP
143  // callback thread. The CUDA/HIP thread signals the std::thread when it is ready to execute the task.
144  // The CUDA/HIP thread is waiting for the std::thread to signal that it is finished executing the task
145  // before it executes the next task in the queue (CUDA/HIP stream).
146  std::thread t(
147  [pCallbackSynchronizationData, task]()
148  {
149  // If the callback has not yet been called, we wait for it.
150  {
151  std::unique_lock<std::mutex> lock(pCallbackSynchronizationData->m_mutex);
152  if(pCallbackSynchronizationData->state != CallbackState::notified)
153  {
154  pCallbackSynchronizationData->m_event.wait(
155  lock,
156  [pCallbackSynchronizationData]()
157  { return pCallbackSynchronizationData->state == CallbackState::notified; });
158  }
159 
160  task();
161 
162  // Notify the waiting CUDA thread.
163  pCallbackSynchronizationData->state = CallbackState::finished;
164  }
165  pCallbackSynchronizationData->m_event.notify_one();
166  });
167 
168  t.detach();
169  }
170  };
