31  //! The CPU boost aligned allocator memory allocation trait specialization.
32  template<typename T, typename TAlignment>
33  struct Malloc<T, AllocCpuAligned<TAlignment>>
34  {
35  ALPAKA_FN_HOST static auto malloc(AllocCpuAligned<TAlignment> const& alloc, std::size_t const& sizeElems)
36  -> T*
37  {
38 #if(defined(ALPAKA_ACC_GPU_CUDA_ENABLED) && BOOST_LANG_CUDA) || (defined(ALPAKA_ACC_GPU_HIP_ENABLED) && BOOST_LANG_HIP)
39  // For CUDA host memory must be aligned to 4 kib to pin it with `cudaHostRegister`,
40  // this was described in older programming guides but was removed later.
41  // From testing with PIConGPU and cuda-memcheck we found out that the alignment is still required.
42  //
43  // For HIP the required alignment is the size of a cache line.
44  // https://rocm-developer-tools.github.io/HIP/group__Memory.html#gab8258f051e1a1f7385f794a15300e674
45  // To avoid issues with HIP(cuda) the alignment will be set also for HIP(clang)
46  // to 4kib.
47  // @todo evaluate requirements when the HIP ecosystem is more stable
48  constexpr size_t minAlignement = 4096;
49 #else
50  constexpr size_t minAlignement = TAlignment::value;
51 #endif
52  alpaka::ignore_unused(alloc);
53  return reinterpret_cast<T*>(
54  core::alignedAlloc(std::max(TAlignment::value, minAlignement), sizeElems * sizeof(T)));
55  }
56  };
