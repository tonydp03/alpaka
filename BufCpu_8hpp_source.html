47  public:

245  struct AsyncBufAlloc<TElem, TDim, TIdx, DevCpu>

247  template<typename TQueue, typename TExtent>
248  ALPAKA_FN_HOST static auto allocAsyncBuf(TQueue queue, TExtent const& extent) -> BufCpu<TElem, TDim, TIdx>
249  {
250  ALPAKA_DEBUG_MINIMAL_LOG_SCOPE;
251 
252  static_assert(
253  std::is_same<Dev<TQueue>, DevCpu>::value,
254  "The BufCpu buffer can only be used with a queue on a DevCpu device!");
255  DevCpu const& dev = getDev(queue);
256 
257  // alpaka::AllocCpuAligned is stateless
258  using Allocator
259  = AllocCpuAligned<std::integral_constant<std::size_t, core::vectorization::defaultAlignment>>;
260  static_assert(std::is_empty_v<Allocator>, "AllocCpuAligned is expected to be stateless");
261  TElem* memPtr
262  = alpaka::malloc<TElem>(Allocator{}, static_cast<std::size_t>(extent::getExtentProduct(extent)));
263  auto deleter = [queue = std::move(queue)](TElem* ptr) mutable
264  {
265  alpaka::enqueue(
266  queue,
267  [ptr, queue]()
268  {
269  // free the memory
270  alpaka::free(Allocator{}, ptr);
271  // keep the queue alive until all memory operations are complete
272  [&queue = std::as_const(queue)]() {}();
273  });
274  };
275 
276  return BufCpu<TElem, TDim, TIdx>(dev, memPtr, std::move(deleter), extent);
277  }

309  //! The BufCpu memory pinning trait specialization.
310  template<typename TElem, typename TDim, typename TIdx>
311  struct Pin<BufCpu<TElem, TDim, TIdx>>
312  {
313  ALPAKA_FN_HOST static auto pin(BufCpu<TElem, TDim, TIdx>& buf) -> void
314  {
315  ALPAKA_DEBUG_MINIMAL_LOG_SCOPE;
316 
317  if(!isPinned(buf))
318  {
319 #if(defined(ALPAKA_ACC_GPU_CUDA_ENABLED) && BOOST_LANG_CUDA) || (defined(ALPAKA_ACC_GPU_HIP_ENABLED) && BOOST_LANG_HIP)
320  if(buf.m_spBufCpuImpl->m_extentElements.prod() != 0)
321  {
322  // - cudaHostRegisterDefault:
323  // See http://cgi.cs.indiana.edu/~nhusted/dokuwiki/doku.php?id=programming:cudaperformance1
324  // - cudaHostRegisterPortable:
325  // The memory returned by this call will be considered as pinned memory by all CUDA contexts,
326  // not just the one that performed the allocation.
327  ALPAKA_UNIFORM_CUDA_HIP_RT_CHECK_IGNORE(
328  ALPAKA_API_PREFIX(HostRegister)(
329  const_cast<void*>(reinterpret_cast<void const*>(getPtrNative(buf))),
330  extent::getExtentProduct(buf) * sizeof(Elem<BufCpu<TElem, TDim, TIdx>>),
331  ALPAKA_API_PREFIX(HostRegisterDefault)),
332  ALPAKA_API_PREFIX(ErrorHostMemoryAlreadyRegistered));
333 
334  buf.m_spBufCpuImpl->m_bPinned = true;
335  }
336 #else
337  static_assert(
338  meta::DependentFalseType<TElem>::value,
339  "Memory pinning of BufCpu is not implemented when CUDA or HIP is not enabled!");
340 #endif
341  }
342  }
343  };
