80  //! The CUDA/HIP RT blocking queue enqueue trait specialization.
81  template<typename TTask>
82  struct Enqueue<QueueUniformCudaHipRtBlocking, TTask>
83  {
84  enum class CallbackState
85  {
86  enqueued,
87  notified,
88  finished,
89  };
90 
91  struct CallbackSynchronizationData : public std::enable_shared_from_this<CallbackSynchronizationData>
92  {
93  std::mutex m_mutex;
94  std::condition_variable m_event;
95  CallbackState state = CallbackState::enqueued;
96  };
97 
98 # if defined(ALPAKA_ACC_GPU_CUDA_ENABLED)
99  static void CUDART_CB
100 # else
101  static void HIPRT_CB
102 # endif
103  uniformCudaHipRtCallback(
104  ALPAKA_API_PREFIX(Stream_t) /*queue*/,
105  ALPAKA_API_PREFIX(Error_t) /*status*/,
106  void* arg)
107  {
108  // explicitly copy the shared_ptr so that this method holds the state even when the executing thread
109  // has already finished.
110  const auto pCallbackSynchronizationData
111  = reinterpret_cast<CallbackSynchronizationData*>(arg)->shared_from_this();
112 
113  // Notify the executing thread.
114  {
115  std::unique_lock<std::mutex> lock(pCallbackSynchronizationData->m_mutex);
116  pCallbackSynchronizationData->state = CallbackState::notified;
117  }
118  pCallbackSynchronizationData->m_event.notify_one();
119 
120  // Wait for the executing thread to finish the task if it has not already finished.
121  std::unique_lock<std::mutex> lock(pCallbackSynchronizationData->m_mutex);
122  if(pCallbackSynchronizationData->state != CallbackState::finished)
123  {
124  pCallbackSynchronizationData->m_event.wait(
125  lock,
126  [pCallbackSynchronizationData]()
127  { return pCallbackSynchronizationData->state == CallbackState::finished; });
128  }
129  }
130 
131  ALPAKA_FN_HOST static auto enqueue(QueueUniformCudaHipRtBlocking& queue, TTask const& task) -> void
132  {
133  auto pCallbackSynchronizationData = std::make_shared<CallbackSynchronizationData>();
134 
135  ALPAKA_UNIFORM_CUDA_HIP_RT_CHECK(ALPAKA_API_PREFIX(StreamAddCallback)(
136  queue.getNativeHandle(),
137  uniformCudaHipRtCallback,
138  pCallbackSynchronizationData.get(),
139  0u));
140 
141  // We start a new std::thread which stores the task to be executed.
142  // This circumvents the limitation that it is not possible to call CUDA/HIP methods within the CUDA/HIP
143  // callback thread. The CUDA/HIP thread signals the std::thread when it is ready to execute the task.
144  // The CUDA/HIP thread is waiting for the std::thread to signal that it is finished executing the task
145  // before it executes the next task in the queue (CUDA/HIP stream).
146  std::thread t(
147  [pCallbackSynchronizationData, task]()
148  {
149  // If the callback has not yet been called, we wait for it.
150  {
151  std::unique_lock<std::mutex> lock(pCallbackSynchronizationData->m_mutex);
152  if(pCallbackSynchronizationData->state != CallbackState::notified)
153  {
154  pCallbackSynchronizationData->m_event.wait(
155  lock,
156  [pCallbackSynchronizationData]()
157  { return pCallbackSynchronizationData->state == CallbackState::notified; });
158  }
159 
160  task();
161 
162  // Notify the waiting CUDA/HIP thread.
163  pCallbackSynchronizationData->state = CallbackState::finished;
164  }
165  pCallbackSynchronizationData->m_event.notify_one();
166  });
167 
168  t.join();
169  }
170  };
