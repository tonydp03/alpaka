43  //! Dynamic block shared memory provider using fixed-size
44  //! member array to allocate memory on the stack or in shared
45  //! memory.
46  template<std::size_t TStaticAllocKiB = ALPAKA_BLOCK_SHARED_DYN_MEMBER_ALLOC_KIB>
47  class alignas(core::vectorization::defaultAlignment) BlockSharedMemDynMember
48  : public concepts::Implements<ConceptBlockSharedDyn, BlockSharedMemDynMember<TStaticAllocKiB>>
49  {
50  public:
51  BlockSharedMemDynMember(std::size_t sizeBytes) : m_dynPitch(getPitch(sizeBytes))
52  {
53  ALPAKA_ASSERT_OFFLOAD(static_cast<std::uint32_t>(sizeBytes) <= staticAllocBytes());
54  }
55 
56  uint8_t* dynMemBegin() const

61  /*! \return the pointer to the begin of data after the portion allocated as dynamical shared memory.

63  uint8_t* staticMemBegin() const
64  {
65  return m_mem.data() + m_dynPitch;
66  }
67 
68  /*! \return the remaining capacity for static block shared memory,

71  std::uint32_t staticMemCapacity() const
72  {

74  }

76  //! \return size of statically allocated memory available for both

78  //! for register efficiency on GPUs
