23  //! Rounds to the next higher power of two (if not already power of two).
24  // Adapted from llvm/ADT/SmallPtrSet.h
25  ALPAKA_FN_INLINE ALPAKA_FN_HOST auto alignedAlloc(size_t alignment, size_t size) -> void*
26  {
27 #if BOOST_OS_WINDOWS
28  return _aligned_malloc(size, alignment);
29 #elif BOOST_OS_MACOS
30  void* ptr = nullptr;
31  posix_memalign(&ptr, alignment, size);
32  return ptr;
33 #else
34  // the amount of bytes to allocate must be a multiple of the alignment
35  size_t sizeToAllocate = ((size + alignment - 1u) / alignment) * alignment;
36  return ::aligned_alloc(alignment, sizeToAllocate);
37 #endif
38  }
