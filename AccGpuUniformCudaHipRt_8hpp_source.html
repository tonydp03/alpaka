113  ALPAKA_FN_HOST static auto getAccDevProps(DevUniformCudaHipRt const& dev) -> AccDevProps<TDim, TIdx>
114  {
115 # ifdef ALPAKA_ACC_GPU_CUDA_ENABLED
116  // Reading only the necessary attributes with cudaDeviceGetAttribute is faster than reading all with
117  // cuda https://devblogs.nvidia.com/cuda-pro-tip-the-fast-way-to-query-device-properties/
118  int multiProcessorCount = {};
119  ALPAKA_UNIFORM_CUDA_HIP_RT_CHECK(cudaDeviceGetAttribute(
120  &multiProcessorCount,
121  cudaDevAttrMultiProcessorCount,
122  dev.getNativeHandle()));
123 
124  int maxGridSize[3] = {};
125  ALPAKA_UNIFORM_CUDA_HIP_RT_CHECK(
126  cudaDeviceGetAttribute(&maxGridSize[0], cudaDevAttrMaxGridDimX, dev.getNativeHandle()));
127  ALPAKA_UNIFORM_CUDA_HIP_RT_CHECK(
128  cudaDeviceGetAttribute(&maxGridSize[1], cudaDevAttrMaxGridDimY, dev.getNativeHandle()));
129  ALPAKA_UNIFORM_CUDA_HIP_RT_CHECK(
130  cudaDeviceGetAttribute(&maxGridSize[2], cudaDevAttrMaxGridDimZ, dev.getNativeHandle()));
131 
132  int maxBlockDim[3] = {};
133  ALPAKA_UNIFORM_CUDA_HIP_RT_CHECK(
134  cudaDeviceGetAttribute(&maxBlockDim[0], cudaDevAttrMaxBlockDimX, dev.getNativeHandle()));
135  ALPAKA_UNIFORM_CUDA_HIP_RT_CHECK(
136  cudaDeviceGetAttribute(&maxBlockDim[1], cudaDevAttrMaxBlockDimY, dev.getNativeHandle()));
137  ALPAKA_UNIFORM_CUDA_HIP_RT_CHECK(
138  cudaDeviceGetAttribute(&maxBlockDim[2], cudaDevAttrMaxBlockDimZ, dev.getNativeHandle()));
139 
140  int maxThreadsPerBlock = {};
141  ALPAKA_UNIFORM_CUDA_HIP_RT_CHECK(
142  cudaDeviceGetAttribute(&maxThreadsPerBlock, cudaDevAttrMaxThreadsPerBlock, dev.getNativeHandle()));
143 
144  int sharedMemSizeBytes = {};
145  ALPAKA_UNIFORM_CUDA_HIP_RT_CHECK(cudaDeviceGetAttribute(
146  &sharedMemSizeBytes,
147  cudaDevAttrMaxSharedMemoryPerBlock,
148  dev.getNativeHandle()));
149 
150  return {// m_multiProcessorCount
151  alpaka::core::clipCast<TIdx>(multiProcessorCount),
152  // m_gridBlockExtentMax
153  getExtentVecEnd<TDim>(Vec<DimInt<3u>, TIdx>(
154  alpaka::core::clipCast<TIdx>(maxGridSize[2u]),
155  alpaka::core::clipCast<TIdx>(maxGridSize[1u]),
156  alpaka::core::clipCast<TIdx>(maxGridSize[0u]))),
157  // m_gridBlockCountMax
158  std::numeric_limits<TIdx>::max(),
159  // m_blockThreadExtentMax
160  getExtentVecEnd<TDim>(Vec<DimInt<3u>, TIdx>(
161  alpaka::core::clipCast<TIdx>(maxBlockDim[2u]),
162  alpaka::core::clipCast<TIdx>(maxBlockDim[1u]),
163  alpaka::core::clipCast<TIdx>(maxBlockDim[0u]))),
164  // m_blockThreadCountMax
165  alpaka::core::clipCast<TIdx>(maxThreadsPerBlock),
166  // m_threadElemExtentMax
167  Vec<TDim, TIdx>::all(std::numeric_limits<TIdx>::max()),
168  // m_threadElemCountMax
169  std::numeric_limits<TIdx>::max(),
170  // m_sharedMemSizeBytes
171  static_cast<size_t>(sharedMemSizeBytes)};
172 
173 # else
174  hipDeviceProp_t hipDevProp;
175  ALPAKA_UNIFORM_CUDA_HIP_RT_CHECK(hipGetDeviceProperties(&hipDevProp, dev.getNativeHandle()));
176 
177  return {// m_multiProcessorCount
178  alpaka::core::clipCast<TIdx>(hipDevProp.multiProcessorCount),
179  // m_gridBlockExtentMax
180  getExtentVecEnd<TDim>(Vec<DimInt<3u>, TIdx>(
181  alpaka::core::clipCast<TIdx>(hipDevProp.maxGridSize[2u]),
182  alpaka::core::clipCast<TIdx>(hipDevProp.maxGridSize[1u]),
183  alpaka::core::clipCast<TIdx>(hipDevProp.maxGridSize[0u]))),
184  // m_gridBlockCountMax
185  std::numeric_limits<TIdx>::max(),
186  // m_blockThreadExtentMax
187  getExtentVecEnd<TDim>(Vec<DimInt<3u>, TIdx>(
188  alpaka::core::clipCast<TIdx>(hipDevProp.maxThreadsDim[2u]),
189  alpaka::core::clipCast<TIdx>(hipDevProp.maxThreadsDim[1u]),
190  alpaka::core::clipCast<TIdx>(hipDevProp.maxThreadsDim[0u]))),
191  // m_blockThreadCountMax
192  alpaka::core::clipCast<TIdx>(hipDevProp.maxThreadsPerBlock),
193  // m_threadElemExtentMax
194  Vec<TDim, TIdx>::all(std::numeric_limits<TIdx>::max()),
195  // m_threadElemCountMax
196  std::numeric_limits<TIdx>::max(),
197  // m_sharedMemSizeBytes
198  static_cast<size_t>(hipDevProp.sharedMemPerBlock)};
