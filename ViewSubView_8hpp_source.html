215  {
216  // \TODO: pre-calculate this pointer for faster execution.
217  return reinterpret_cast<TElem*>(
218  reinterpret_cast<std::uint8_t*>(alpaka::getPtrNative(view.m_viewParentView))
219  + pitchedOffsetBytes(view, IdxSequence()));
220  }
221 
222  private:
223  //! For a 3D vector this calculates:
224  //!
225  //! getOffset<0u>(view) * getPitchBytes<1u>(view)
226  //! + getOffset<1u>(view) * getPitchBytes<2u>(view)
227  //! + getOffset<2u>(view) * getPitchBytes<3u>(view)
228  //! while getPitchBytes<3u>(view) is equivalent to sizeof(TElem)
229  template<typename TView, std::size_t... TIndices>
230  ALPAKA_FN_HOST static auto pitchedOffsetBytes(TView const& view, std::index_sequence<TIndices...> const&)
231  -> TIdx
232  {
233  return meta::foldr(std::plus<TIdx>(), pitchedOffsetBytesDim<TIndices>(view)...);
234  }
235  template<std::size_t Tidx, typename TView>
236  ALPAKA_FN_HOST static auto pitchedOffsetBytesDim(TView const& view) -> TIdx
237  {
238  return getOffset<Tidx>(view) * getPitchBytes<Tidx + 1u>(view);
239  }
