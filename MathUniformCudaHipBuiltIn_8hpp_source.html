217  struct Abs<AbsUniformCudaHipBuiltIn, TArg, std::enable_if_t<std::is_signed_v<TArg>>>
218  {

220  {

222  return ::fabsf(arg);

225  else if constexpr(is_decayed_v<TArg, int>)
226  return ::abs(arg);

228  return ::labs(arg);

231  else
232  static_assert(!sizeof(TArg), "Unsupported data type");
233 
234  ALPAKA_UNREACHABLE(TArg{});
235  }
