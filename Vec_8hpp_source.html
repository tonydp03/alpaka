39  //! Single value constructor helper.
40  ALPAKA_NO_HOST_ACC_WARNING
41  template<
42  typename TDim,
43  template<std::size_t>
44  class TTFnObj,
45  typename... TArgs,
46  typename TIdxSize,
47  TIdxSize... TIndices>
48  ALPAKA_FN_HOST_ACC auto createVecFromIndexedFnArbitrary(
49  std::integer_sequence<TIdxSize, TIndices...> const& indices,
50  TArgs&&... args)
51  {
52  alpaka::ignore_unused(indices);
53 
54  return Vec<TDim, decltype(TTFnObj<0>::create(std::forward<TArgs>(args)...))>(
55  (TTFnObj<TIndices>::create(std::forward<TArgs>(args)...))...);
56  }
57  //! Creator using func<idx>(args...) to initialize all values of the vector.
58  //! The idx is in the range [0, TDim].
59  ALPAKA_NO_HOST_ACC_WARNING
60  template<typename TDim, template<std::size_t> class TTFnObj, typename... TArgs>
61  ALPAKA_FN_HOST_ACC auto createVecFromIndexedFn(TArgs&&... args)
62  {
63  using IdxSequence = std::make_integer_sequence<typename TDim::value_type, TDim::value>;
64  return createVecFromIndexedFnArbitrary<TDim, TTFnObj>(IdxSequence(), std::forward<TArgs>(args)...);
65  }
66  //! Creator using func<idx>(args...) to initialize all values of the vector.
67  //! The idx is in the range [TIdxOffset, TIdxOffset + TDim].
68  ALPAKA_NO_HOST_ACC_WARNING
69  template<typename TDim, template<std::size_t> class TTFnObj, typename TIdxOffset, typename... TArgs>
70  ALPAKA_FN_HOST_ACC auto createVecFromIndexedFnOffset(TArgs&&... args)
71  {
72  using IdxSubSequenceSigned = meta::MakeIntegerSequenceOffset<std::intmax_t, TIdxOffset::value, TDim::value>;
73  using IdxSubSequence = meta::ConvertIntegerSequence<typename TIdxOffset::value_type, IdxSubSequenceSigned>;
74  return createVecFromIndexedFnArbitrary<TDim, TTFnObj>(IdxSubSequence(), std::forward<TArgs>(args)...);
75  }

77  //! A n-dimensional vector.
78  template<typename TDim, typename TVal>
79  class Vec final
80  {
81  public:
82  static_assert(TDim::value >= 0u, "Invalid dimensionality");
83 
84  using Dim = TDim;
85  using Val = TVal;
86 
87  private:
88  //! A sequence of integers from 0 to dim-1.
89  //! This can be used to write compile time indexing algorithms.
90  using IdxSequence = std::make_integer_sequence<std::size_t, TDim::value>;
91 
92  public:
93  // The default constructor is only available when the vector is zero-dimensional.
94  ALPAKA_NO_HOST_ACC_WARNING
95  template<bool B = (TDim::value == 0u), typename = std::enable_if_t<B>>
96  ALPAKA_FN_HOST_ACC Vec() : m_data{static_cast<TVal>(0u)}
97  {
98  }
99 
100 
101  //! Value constructor.
102  //! This constructor is only available if the number of parameters matches the vector idx.
103  ALPAKA_NO_HOST_ACC_WARNING
104  template<
105  typename TArg0,
106  typename... TArgs,
107  typename = std::enable_if_t<
108  // There have to be dim arguments.
109  (sizeof...(TArgs) + 1 == TDim::value) && (std::is_same<TVal, std::decay_t<TArg0>>::value)>>
110  ALPAKA_FN_HOST_ACC Vec(TArg0&& arg0, TArgs&&... args)
111  : m_data{std::forward<TArg0>(arg0), std::forward<TArgs>(args)...}
112  {
113  }
114 
115  private:
116  //! A function object that returns the given value for each index.
117  template<std::size_t Tidx>
118  struct CreateSingleVal
119  {
120  ALPAKA_NO_HOST_ACC_WARNING
121  ALPAKA_FN_HOST_ACC static auto create(TVal const& val) -> TVal
122  {
123  return val;
124  }
125  };
126 
127  public:
128  //! \brief Single value constructor.
129  //!
130  //! Creates a vector with all values set to val.
131  //! \param val The initial value.
132  ALPAKA_NO_HOST_ACC_WARNING
133  ALPAKA_FN_HOST_ACC static auto all(TVal const& val) -> Vec<TDim, TVal>
134  {
135  return createVecFromIndexedFn<TDim, CreateSingleVal>(val);
136  }
137  //! Zero value constructor.
138  ALPAKA_NO_HOST_ACC_WARNING
139  ALPAKA_FN_HOST_ACC static auto zeros() -> Vec<TDim, TVal>
140  {
141  return all(static_cast<TVal>(0));
142  }
143  //! One value constructor.
144  ALPAKA_NO_HOST_ACC_WARNING
145  ALPAKA_FN_HOST_ACC static auto ones() -> Vec<TDim, TVal>
146  {
147  return all(static_cast<TVal>(1));
148  }
149 
150  //! Value reference accessor at the given non-unsigned integer index.
151  //! \return A reference to the value at the given index.
152  ALPAKA_NO_HOST_ACC_WARNING
153  template<typename TIdx, typename = std::enable_if_t<std::is_integral<TIdx>::value>>
154  ALPAKA_FN_HOST_ACC auto operator[](TIdx const iIdx) -> TVal&
155  {
156  core::assertValueUnsigned(iIdx);
157  auto const idx = static_cast<typename TDim::value_type>(iIdx);
158  core::assertGreaterThan<TDim>(idx);
159  return m_data[idx];
160  }
161 
162  //! Value accessor at the given non-unsigned integer index.
163  //! \return The value at the given index.
164  ALPAKA_NO_HOST_ACC_WARNING
165  template<typename TIdx, typename = std::enable_if_t<std::is_integral<TIdx>::value>>
166  ALPAKA_FN_HOST_ACC auto operator[](TIdx const iIdx) const -> TVal
167  {
168  core::assertValueUnsigned(iIdx);
169  auto const idx = static_cast<typename TDim::value_type>(iIdx);
170  core::assertGreaterThan<TDim>(idx);
171  return m_data[idx];
172  }
173 
174  ALPAKA_NO_HOST_ACC_WARNING
175  ALPAKA_FN_HOST_ACC auto operator==(Vec const& rhs) const -> bool
176  {
177  for(typename TDim::value_type i(0); i < TDim::value; ++i)
178  {
179  if((*this)[i] != rhs[i])
180  {
181  return false;
182  }
183  }
184  return true;
185  }
186  ALPAKA_NO_HOST_ACC_WARNING
187  ALPAKA_FN_HOST_ACC auto operator!=(Vec const& rhs) const -> bool
188  {
189  return !((*this) == rhs);
190  }
191  ALPAKA_NO_HOST_ACC_WARNING
192  template<typename TFnObj, std::size_t... TIndices>
193  ALPAKA_FN_HOST_ACC auto foldrByIndices(
194  TFnObj const& f,
195  std::integer_sequence<std::size_t, TIndices...> const& indices) const
196  {
197  alpaka::ignore_unused(indices);
198 
199  return meta::foldr(f, ((*this)[TIndices])...);
200  }
201  ALPAKA_NO_HOST_ACC_WARNING
202  template<typename TFnObj>
203  ALPAKA_FN_HOST_ACC auto foldrAll(TFnObj const& f) const
204  {
205  return foldrByIndices(f, IdxSequence());
206  }
207 // suppress strange warning produced by nvcc+MSVC in release mode
208 #if BOOST_COMP_MSVC || defined(BOOST_COMP_MSVC_EMULATED)
209 # pragma warning(push)
210 # pragma warning(disable : 4702) // unreachable code
211 #endif
212  //! \return The product of all values.
213  ALPAKA_NO_HOST_ACC_WARNING
214  ALPAKA_FN_HOST_ACC auto prod() const -> TVal
215  {
216  return foldrAll(std::multiplies<TVal>());
217  }
218 #if BOOST_COMP_MSVC || defined(BOOST_COMP_MSVC_EMULATED)
219 # pragma warning(pop)
220 #endif
221  //! \return The sum of all values.
222  ALPAKA_NO_HOST_ACC_WARNING
223  ALPAKA_FN_HOST_ACC auto sum() const -> TVal
224  {
225  return foldrAll(std::plus<TVal>());
226  }
227  //! \return The min of all values.
228  ALPAKA_NO_HOST_ACC_WARNING
229  ALPAKA_FN_HOST_ACC auto min() const -> TVal
230  {
231  return foldrAll(meta::min<TVal>());
232  }
233  //! \return The max of all values.
234  ALPAKA_NO_HOST_ACC_WARNING
235  ALPAKA_FN_HOST_ACC auto max() const -> TVal
236  {
237  return foldrAll(meta::max<TVal>());
238  }
239  //! \return The index of the minimal element.
240  ALPAKA_FN_HOST auto minElem() const -> typename TDim::value_type
241  {
242  return static_cast<typename TDim::value_type>(
243  std::distance(std::begin(m_data), std::min_element(std::begin(m_data), std::end(m_data))));
244  }
245  //! \return The index of the maximal element.
246  ALPAKA_FN_HOST auto maxElem() const -> typename TDim::value_type
247  {
248  return static_cast<typename TDim::value_type>(
249  std::distance(std::begin(m_data), std::max_element(std::begin(m_data), std::end(m_data))));
250  }
251 
252  template<size_t I>
253  ALPAKA_FN_HOST_ACC auto get() -> TVal&
254  {
255  return (*this)[I];
256  }
257 
258  template<size_t I>
259  ALPAKA_FN_HOST_ACC auto get() const -> TVal
260  {
261  return (*this)[I];
262  }
263 
264  private:
265  // Zero sized arrays are not allowed, therefore zero-dimensional vectors have one member.
266  TVal m_data[TDim::value == 0u ? 1u : TDim::value];
267  };

419  //! Specialization for selecting a sub-vector.

421  struct SubVecFromIndices<
422  Vec<TDim, TVal>,
423  std::integer_sequence<std::size_t, TIndices...>,

425  std::integer_sequence<std::size_t, TIndices...>,

428  ALPAKA_NO_HOST_ACC_WARNING
