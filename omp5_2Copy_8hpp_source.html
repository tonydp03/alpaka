36  //! The Omp5 memory copy trait.
37  template<typename TDim, typename TViewDst, typename TViewSrc, typename TExtent>
38  struct TaskCopyOmp5
39  {
40  static_assert(!std::is_const<TViewDst>::value, "The destination view can not be const!");
41 
42  static_assert(
43  Dim<TViewSrc>::value == TDim::value,
44  "The source view is required to have dimensionality TDim!");
45  static_assert(
46  Dim<TViewDst>::value == Dim<TViewSrc>::value,
47  "The source and the destination view are required to have the same dimensionality!");
48  static_assert(
49  Dim<TViewDst>::value == Dim<TExtent>::value,
50  "The views and the extent are required to have the same dimensionality!");
51  // TODO: Maybe check for Idx of TViewDst and TViewSrc to have greater or equal range than TExtent.
52  static_assert(
53  std::is_same<Elem<TViewDst>, typename std::remove_const<Elem<TViewSrc>>::type>::value,
54  "The source and the destination view are required to have the same element type!");
55 
56  using Idx = alpaka::Idx<TExtent>;
57 
58  ALPAKA_FN_HOST TaskCopyOmp5(
59  TViewDst& viewDst,
60  TViewSrc const& viewSrc,
61  TExtent const& extent,
62  int const& iDstDevice,
63  int const& iSrcDevice)
64  : m_iDstDevice(iDstDevice)
65  , m_iSrcDevice(iSrcDevice)
66  , m_extent(castVec<size_t>(extent::getExtentVec(extent)))
67  , m_dstPitchBytes(castVec<size_t>(getPitchBytesVec(viewDst)))
68  , m_srcPitchBytes(castVec<size_t>(getPitchBytesVec(viewSrc)))
69  , m_dstMemNative(reinterpret_cast<void*>(getPtrNative(viewDst)))
70  , m_srcMemNative(reinterpret_cast<void const*>(getPtrNative(viewSrc)))
71  {
72 # if ALPAKA_DEBUG >= ALPAKA_DEBUG_FULL
73  const auto dstExtent(castVec<size_t>(extent::getExtentVec(viewDst)));
74  const auto srcExtent(castVec<size_t>(extent::getExtentVec(viewSrc)));
75  for(auto i = static_cast<decltype(TDim::value)>(0u); i < TDim::value; ++i)
76  {
77  ALPAKA_ASSERT(m_extent[i] <= dstExtent[i]);
78  ALPAKA_ASSERT(m_extent[i] <= srcExtent[i]);
79  }
80  std::cout << "TaskCopyOmp5<" << TDim::value << ",...>::ctor\tsrcExtent=" << srcExtent
81  << ", dstExtent=" << dstExtent << std::endl;
82 # endif
83  }
84 
85 # if ALPAKA_DEBUG >= ALPAKA_DEBUG_FULL
86  ALPAKA_FN_HOST auto printDebug() const -> void
87  {
88  std::cout << __func__ << " ddev: " << m_iDstDevice << " ew: " << m_extent
89  << " dptr: " << m_dstMemNative << " sdev: " << m_iSrcDevice << " sptr: " << m_srcMemNative
90  << std::endl;
91  }
92 # endif
93  int m_iDstDevice;
94  int m_iSrcDevice;
95  Vec<TDim, size_t> m_extent;
96  Vec<TDim, size_t> m_dstPitchBytes;
97  Vec<TDim, size_t> m_srcPitchBytes;
98  void* m_dstMemNative;
99  void const* m_srcMemNative;
100 
101  //! Executes the kernel function object.
102  ALPAKA_FN_HOST auto operator()() const -> void
103  {
104  ALPAKA_DEBUG_FULL_LOG_SCOPE;
105 
106 # if ALPAKA_DEBUG >= ALPAKA_DEBUG_FULL
107  printDebug();
108 # endif
109  constexpr auto lastDim = TDim::value - 1;
110 
111  if(m_extent.prod() > 0)
112  {
113  // offsets == 0 by ptr shift (?)
114  auto dstOffset(Vec<TDim, size_t>::zeros());
115  auto srcOffset(Vec<TDim, size_t>::zeros());
116 
117  auto dstExtentFull(Vec<TDim, size_t>::zeros());
118  auto srcExtentFull(Vec<TDim, size_t>::zeros());
119 
120  const size_t elementSize
121  = (m_dstPitchBytes[0] % sizeof(Elem<TViewDst>) || m_srcPitchBytes[0] % sizeof(Elem<TViewDst>))
122  ? 1
123  : sizeof(Elem<TViewDst>);
124 
125  dstExtentFull[lastDim] = m_dstPitchBytes[lastDim] / elementSize;
126  srcExtentFull[lastDim] = m_srcPitchBytes[lastDim] / elementSize;
127  for(int i = lastDim - 1; i >= 0; --i)
128  {
129  dstExtentFull[i] = m_dstPitchBytes[i] / m_dstPitchBytes[i + 1];
130  srcExtentFull[i] = m_srcPitchBytes[i] / m_srcPitchBytes[i + 1];
131  }
132 
133  ALPAKA_OMP5_CHECK(omp_target_memcpy_rect(
134  m_dstMemNative,
135  const_cast<void*>(m_srcMemNative),
136  sizeof(Elem<TViewDst>),
137  TDim::value,
138  reinterpret_cast<size_t const*>(&m_extent),
139  reinterpret_cast<size_t const*>(&dstOffset),
140  reinterpret_cast<size_t const*>(&srcOffset),
141  reinterpret_cast<size_t const*>(&dstExtentFull),
142  reinterpret_cast<size_t const*>(&srcExtentFull),
143  m_iDstDevice,
144  m_iSrcDevice));
145  }
146  }
147  };

188  }

194  << " ewb: " << m_extentWidthBytes << " dw: " << m_dstWidth << " dptr: " << m_dstMemNative
195  << " sdev: " << m_iSrcDevice << " sw: " << m_srcWidth << " sptr: " << m_srcMemNative
196  << std::endl;
197  }
198 # endif

203  Idx m_dstWidth;
204  Idx m_srcWidth;
205 # endif

207  void* m_dstMemNative;
208  void const* m_srcMemNative;
209

220  return;
