41  //! The OMP5 memory buffer detail.
42  template<typename TElem, typename TDim, typename TIdx>
43  class BufOmp5Impl
44  {
45  static_assert(
46  !std::is_const<TElem>::value,
47  "The elem type of the buffer can not be const because the C++ Standard forbids containers of const "
48  "elements!");
49  static_assert(!std::is_const<TIdx>::value, "The idx type of the buffer can not be const!");
50 
51  private:
52  using Elem = TElem;
53  using Dim = TDim;
54  //! Calculate the pitches purely from the extents.
55  template<typename TExtent>
56  ALPAKA_FN_HOST static auto calculatePitchesFromExtents(TExtent const& extent) -> Vec<TDim, TIdx>
57  {
58  Vec<TDim, TIdx> pitchBytes(Vec<TDim, TIdx>::all(0));
59  pitchBytes[TDim::value - 1u] = extent[TDim::value - 1u] * static_cast<TIdx>(sizeof(TElem));
60  for(TIdx i = TDim::value - 1u; i > static_cast<TIdx>(0u); --i)
61  {
62  pitchBytes[i - 1] = extent[i - 1] * pitchBytes[i];
63  }
64  return pitchBytes;
65  }
66 
67  public:
68  //! Constructor
69  template<typename TExtent>
70  ALPAKA_FN_HOST BufOmp5Impl(DevOmp5 const& dev, TElem* const pMem, TExtent const& extent)
71  : m_dev(dev)
72  , m_extentElements(extent::getExtentVecEnd<TDim>(extent))
73  , m_pitchBytes(calculatePitchesFromExtents(m_extentElements))
74  , m_pMem(pMem)
75  {
76  ALPAKA_DEBUG_MINIMAL_LOG_SCOPE;
77 
78  static_assert(
79  TDim::value == alpaka::Dim<TExtent>::value,
80  "The dimensionality of TExtent and the dimensionality of the TDim template parameter have to be "
81  "identical!");
82  static_assert(
83  std::is_same<TIdx, Idx<TExtent>>::value,
84  "The idx type of TExtent and the TIdx template parameter have to be identical!");
85  }
86 
87  public:
88  DevOmp5 m_dev;
89  Vec<TDim, TIdx> m_extentElements;
90  Vec<TDim, TIdx> m_pitchBytes;
91  TElem* m_pMem;
92 
93  BufOmp5Impl(BufOmp5Impl&&) = default;
94  auto operator=(BufOmp5Impl&&) -> BufOmp5Impl& = default;
95  ~BufOmp5Impl()
96  {
97  omp_target_free(m_pMem, m_dev.m_spDevOmp5Impl->getNativeHandle());
