38  //! The CUDA/HIP RT device event implementation.
39  class EventUniformCudaHipImpl final
40  {
41  public:
42  ALPAKA_FN_HOST EventUniformCudaHipImpl(DevUniformCudaHipRt const& dev, bool bBusyWait)
43  : m_dev(dev)
44  , m_UniformCudaHipEvent()
45  {
46  ALPAKA_DEBUG_MINIMAL_LOG_SCOPE;
47 
48  // Set the current device.
49  ALPAKA_UNIFORM_CUDA_HIP_RT_CHECK(ALPAKA_API_PREFIX(SetDevice)(m_dev.getNativeHandle()));
50 
51  // Create the event on the current device with the specified flags. Valid flags include:
52  // - cuda/hip-EventDefault: Default event creation flag.
53  // - cuda/hip-EventBlockingSync : Specifies that event should use blocking synchronization.
54  // A host thread that uses cuda/hip-EventSynchronize() to wait on an event created with this flag
55  // will block until the event actually completes.
56  // - cuda/hip-EventDisableTiming : Specifies that the created event does not need to record timing
57  // data.
58  // Events created with this flag specified and the cuda/hip-EventBlockingSync flag not specified
59  // will provide the best performance when used with cudaStreamWaitEvent() and cudaEventQuery().
60  ALPAKA_UNIFORM_CUDA_HIP_RT_CHECK(ALPAKA_API_PREFIX(EventCreateWithFlags)(
61  &m_UniformCudaHipEvent,
62  (bBusyWait ? ALPAKA_API_PREFIX(EventDefault) : ALPAKA_API_PREFIX(EventBlockingSync))
63  | ALPAKA_API_PREFIX(EventDisableTiming)));
64  }
65  EventUniformCudaHipImpl(EventUniformCudaHipImpl const&) = delete;
66  auto operator=(EventUniformCudaHipImpl const&) -> EventUniformCudaHipImpl& = delete;
67  ALPAKA_FN_HOST ~EventUniformCudaHipImpl()
68  {
69  ALPAKA_DEBUG_MINIMAL_LOG_SCOPE;
70 
71  // In case event has been recorded but has not yet been completed when cuda/hip-EventDestroy() is
72  // called, the function will return immediately and the resources associated with event will be
73  // released automatically once the device has completed event.
74  // -> No need to synchronize here.
75  ALPAKA_UNIFORM_CUDA_HIP_RT_CHECK_NOEXCEPT(ALPAKA_API_PREFIX(EventDestroy)(m_UniformCudaHipEvent));
76  }
77 
78  [[nodiscard]] auto getNativeHandle() const noexcept
79  {
80  return m_UniformCudaHipEvent;
81  }
82 
83  public:
84  DevUniformCudaHipRt const m_dev; //!< The device this event is bound to.
85 
86  private:
87  ALPAKA_API_PREFIX(Event_t) m_UniformCudaHipEvent;
88  };
89  } // namespace uniform_cuda_hip::detail
