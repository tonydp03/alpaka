35  //! The CUDA/HIP RT blocking queue implementation.
36  class QueueUniformCudaHipRtImpl final
37  {
38  public:
39  ALPAKA_FN_HOST QueueUniformCudaHipRtImpl(DevUniformCudaHipRt const& dev)
40  : m_dev(dev)
41  , m_UniformCudaHipQueue()
42  {
43  ALPAKA_DEBUG_MINIMAL_LOG_SCOPE;
44 
45  // Set the current device.
46  ALPAKA_UNIFORM_CUDA_HIP_RT_CHECK(ALPAKA_API_PREFIX(SetDevice)(m_dev.m_iDevice));
47 
48  // - [cuda/hip]StreamDefault: Default queue creation flag.
49  // - [cuda/hip]StreamNonBlocking: Specifies that work running in the created queue may run
50  // concurrently with work in queue 0 (the NULL queue),
51  // and that the created queue should perform no implicit synchronization with queue 0.
52  // Create the queue on the current device.
53  // NOTE: [cuda/hip]StreamNonBlocking is required to match the semantic implemented in the alpaka
54  // CPU queue. It would be too much work to implement implicit default queue synchronization on CPU.
55 
56  ALPAKA_UNIFORM_CUDA_HIP_RT_CHECK(ALPAKA_API_PREFIX(
57  StreamCreateWithFlags)(&m_UniformCudaHipQueue, ALPAKA_API_PREFIX(StreamNonBlocking)));
58  }
59  QueueUniformCudaHipRtImpl(QueueUniformCudaHipRtImpl&&) = default;
60  auto operator=(QueueUniformCudaHipRtImpl&&) -> QueueUniformCudaHipRtImpl& = delete;
61  ALPAKA_FN_HOST ~QueueUniformCudaHipRtImpl()
62  {
63  ALPAKA_DEBUG_MINIMAL_LOG_SCOPE;
64 
65  // In case the device is still doing work in the queue when cuda/hip-StreamDestroy() is called, the
66  // function will return immediately and the resources associated with queue will be released
67  // automatically once the device has completed all work in queue.
68  // -> No need to synchronize here.
69  ALPAKA_UNIFORM_CUDA_HIP_RT_CHECK(ALPAKA_API_PREFIX(StreamDestroy)(m_UniformCudaHipQueue));
70  }
71 
72  public:
73  DevUniformCudaHipRt const m_dev; //!< The device this queue is bound to.
74  ALPAKA_API_PREFIX(Stream_t) m_UniformCudaHipQueue;
75  };
