54  //! The GPU random number normal distribution.
55  template<typename T>
56  class UniformUint
57  {
58  static_assert(std::is_integral<T>::value, "Return type of UniformUint must be integral.");
59 
60  public:
61  UniformUint() = default;
62 
63  template<typename TEngine>
64  ALPAKA_FN_HOST_ACC auto operator()(TEngine& engine) -> T
65  {
66  using BitsT = typename TEngine::result_type;
67  T ret = 0;
68  constexpr auto N = sizeof(T) / sizeof(BitsT);
69  for(unsigned int a = 0; a < N; ++a)
70  {
71  ret
72  ^= (static_cast<T>(engine())
73  << (sizeof(BitsT) * std::numeric_limits<unsigned char>::digits * a));
74  }
75  return ret;
76  }
77  };

79  //! The GPU random number uniform distribution.
80  template<typename T>
81  class UniformReal
82  {
83  static_assert(
84  std::is_floating_point<T>::value,
85  "Return type of UniformReal must be floating point.");
86 
87  using BitsT = typename detail::BitsType<T>::type;
88 
89  public:
90  UniformReal() = default;
91 
92  template<typename TEngine>
93  ALPAKA_FN_HOST_ACC auto operator()(TEngine& engine) -> T
94  {
95  constexpr BitsT limit = static_cast<BitsT>(1) << std::numeric_limits<T>::digits;
96  const BitsT b = UniformUint<BitsT>()(engine);
97  const auto ret = static_cast<T>(b & (limit - 1)) / limit;
98  return ret;
99  }

102  /*! The GPU random number normal distribution.
103  *
104  * \note
105  * This type contains state and is not thread-safe: To be used
106  * per thread, not shared.
107  *
108  * \note When reproducibility is a concern, each instance of
109  * this class should be used with only on random engine
110  * instance, or two consecutive number should be generated with
111  * each engine used. This is due to the implicit caching of one
112  * Gaussian random number.
113  */
114  template<typename Acc, typename T>
115  class NormalReal
116  {
117  static_assert(
118  std::is_floating_point<T>::value,
119  "Return type of NormalReal must be floating point.");
120 
121  const Acc& m_acc;
122  T m_cache = std::numeric_limits<T>::quiet_NaN();
123 
124  public:
125  /*! \warning Retains a reference to \p acc, thus must not
126  * outlive it.
127  */
128  NormalReal(const Acc& acc) : m_acc(acc)
129  {
130  }
131 
132  NormalReal(const NormalReal& o) = delete;
133 
134  //! The move ctor clears `m_cache` of source.
135  //! \todo This is to be deleted when moving to C++17
136  NormalReal(NormalReal&& o) : m_acc(o.m_acc), m_cache(o.m_cache)
137  {
138  o.m_cache = std::numeric_limits<T>::quiet_NaN();
139  }
140 
141  template<typename TEngine>
142  ALPAKA_FN_HOST_ACC auto operator()(TEngine& engine) -> T
143  {
144  constexpr T sigma = 1., mu = 0.;
145  if(math::isnan(m_acc, m_cache))
146  {
147  UniformReal<T> uni;
148 
149  T u1, u2;
150  do
151  {
152  u1 = uni(engine);
153  u2 = uni(engine);
154  } while(u1 <= std::numeric_limits<T>::epsilon());
155 
156  // compute z0 and z1
157  const T mag = sigma * math::sqrt(m_acc, static_cast<T>(-2.) * math::log(m_acc, u1));
158  constexpr T twoPi = static_cast<T>(2. * M_PI);
159  // getting two normal number out of this, store one for later
160  m_cache = mag * static_cast<T>(math::cos(m_acc, twoPi * u2)) + mu;
161 
162  return mag * static_cast<T>(math::sin(m_acc, twoPi * u2)) + mu;
163  }
164  else
165  {
166  const T ret = m_cache;
167  m_cache = std::numeric_limits<T>::quiet_NaN();
168  return ret;
169  }
170  }
171  };
