50  //! The CUDA/HIP memory buffer.
51  template<typename TElem, typename TDim, typename TIdx>
52  class BufUniformCudaHipRt
53  {
54  static_assert(
55  !std::is_const<TElem>::value,
56  "The elem type of the buffer can not be const because the C++ Standard forbids containers of const "
57  "elements!");
58  static_assert(!std::is_const<TIdx>::value, "The idx type of the buffer can not be const!");
59 
60  private:
61  using Elem = TElem;
62  using Dim = TDim;
63 
64  public:
65  //! Constructor
66  template<typename TExtent, typename Deleter>
67  ALPAKA_FN_HOST BufUniformCudaHipRt(
68  DevUniformCudaHipRt const& dev,
69  TElem* const pMem,
70  Deleter deleter,
71  TIdx const& pitchBytes,
72  TExtent const& extent)
73  : m_dev(dev)
74  , m_extentElements(extent::getExtentVecEnd<TDim>(extent))
75  , m_spMem(pMem, std::move(deleter))
76  , m_pitchBytes(pitchBytes)
77  {
78  ALPAKA_DEBUG_MINIMAL_LOG_SCOPE;
79 
80  static_assert(
81  TDim::value == alpaka::Dim<TExtent>::value,
82  "The dimensionality of TExtent and the dimensionality of the TDim template parameter have to be "
83  "identical!");
84  static_assert(
85  std::is_same<TIdx, Idx<TExtent>>::value,
86  "The idx type of TExtent and the TIdx template parameter have to be identical!");
87  }
88 
89  public:
90  DevUniformCudaHipRt m_dev;
91  Vec<TDim, TIdx> m_extentElements;
92  std::shared_ptr<TElem> m_spMem;
93  TIdx m_pitchBytes;
94  };
